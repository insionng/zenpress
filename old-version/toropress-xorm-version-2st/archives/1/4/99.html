golang 接口使用

May 13 , 2013 at 07:15 am - - 203点击 - 0 评论

原文

 

第一种

package main  
  
import "fmt"  
  
type Writer interface {  
    Write(data []byte) (int, error)  
}  
  
type writerImpl struct {  
}  
  
func (w *writerImpl) Write(data []byte) (int, error) {  
    return len(data), nil  
}  
  
type Worker struct {  
    Writer  
}  
  
func main() {  
    worker := Worker{&writerImpl{}}  
  
    // using by inteface  
    n, err := worker.Write(nil)  
    if err != nil {  
        panic(err)  
    }  
  
    fmt.Println(n)  
}

import "fmt"

}

    return len(data), nil  
}

    // using by inteface
